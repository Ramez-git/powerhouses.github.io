Full code for plotting all three graphs:

import numpy as np
import matplotlib.pyplot as plt
import pandas as pd

file_name = "filename.csv"
dataset = pd.read_csv(file_name)

dependent_variable_O_column_name = "position"
position = dataset[dependent_variable_O_column_name]

independent_variable_column_name= "time"

t = dataset[independent_variable_column_name]

# Compute velocity using numerical differentiation
velocity = np.gradient(position, t)

# Compute acceleration using numerical differentiation
acceleration = np.gradient(velocity, t)

# Save data to a CSV file
data = pd.DataFrame({'Time (s)': t, 'Position (m)': position, 
                     'Velocity (m/s)': velocity, 'Acceleration (m/s²)': acceleration})
data.to_csv("motion_data.csv", index=False)

# Plot results
fig, axes = plt.subplots(3, 1, figsize=(8, 10), sharex=True)

axes[0].plot(t, position, label="Position (m)", color='b')
axes[0].set_ylabel("Position (m)")
axes[0].legend()
axes[0].grid()

axes[1].plot(t, velocity, label="Velocity (m/s)", color='g')
axes[1].set_ylabel("Velocity (m/s)")
axes[1].legend()
axes[1].grid()

axes[2].plot(t, acceleration, label="Acceleration (m/s²)", color='r')
axes[2].set_xlabel("Time (s)")
axes[2].set_ylabel("Acceleration (m/s²)")
axes[2].legend()
axes[2].grid()

plt.show()

print("Data saved to 'motion_data.csv'")
